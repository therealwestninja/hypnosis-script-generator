      req.onupgradeneeded = e => {

        if (!db.objectStoreNames.contains(IDB_STORE)) db.createObjectStore(IDB_STORE);
      };
      req.onsuccess = e => {
        _idb = e.target.result;
        _idb.onclose = () => { _idb = null; _idbOpenP = null; };
        resolve(_idb);
      };
      req.onerror = e => { _idbOpenP = null; reject(e.target.error ?? new Error('IDB open failed')); };

  function goToStep(n) {
    if (n < 1 || n > TOTAL_STEPS) return;
    // Forward direction: validate current step
    if (n > state.currentStep) {
      for (let i = state.currentStep; i < n; i++) {
        if (!validateStep(i)) { return; }
        state.stepsCompleted[i] = true;
      }
    }
    state.currentStep = n;
    updateStepUI();
    if (n === 5) { refreshSummary(); }
    window.scrollTo({ top: 0, behavior: 'smooth' });
  }

  class AudioDrone {
    constructor() { this.ctx = null; this.master = null; this.nodes = []; this.running = false; }
    async start(volume = 0.35) {
      if (this.running) return;
      this.ctx = new (window.AudioContext || window.webkitAudioContext)();
      if (this.ctx.state === 'suspended') await this.ctx.resume();
      this.master = this.ctx.createGain(); this.master.gain.value = 0; this.master.connect(this.ctx.destination);
      const bass = this.ctx.createOscillator(); bass.type = 'sine'; bass.frequency.value = 70;
      const bassGain = this.ctx.createGain(); bassGain.gain.value = 0.4; bass.connect(bassGain).connect(this.master); bass.start(); this.nodes.push(bass, bassGain);
      const harm = this.ctx.createOscillator(); harm.type = 'sine'; harm.frequency.value = 141;
      const harmGain = this.ctx.createGain(); harmGain.gain.value = 0.18; harm.connect(harmGain).connect(this.master); harm.start(); this.nodes.push(harm, harmGain);
      const bufferSize = 2 * this.ctx.sampleRate;
      const noiseBuffer = this.ctx.createBuffer(1, bufferSize, this.ctx.sampleRate);
      const data = noiseBuffer.getChannelData(0); let lastOut = 0;
      for (let i = 0; i < bufferSize; i++) { const white = Math.random() * 2 - 1; data[i] = (lastOut + (0.02 * white)) / 1.02; lastOut = data[i]; data[i] *= 3.5; }
      const noise = this.ctx.createBufferSource(); noise.buffer = noiseBuffer; noise.loop = true;
      const nf = this.ctx.createBiquadFilter(); nf.type = 'lowpass'; nf.frequency.value = 280;
      const ng = this.ctx.createGain(); ng.gain.value = 0.25;
      noise.connect(nf).connect(ng).connect(this.master); noise.start(); this.nodes.push(noise, nf, ng);
      const lfo = this.ctx.createOscillator(); lfo.frequency.value = 0.07;
      const lg = this.ctx.createGain(); lg.gain.value = 0.08;
      lfo.connect(lg).connect(bassGain.gain); lfo.start(); this.nodes.push(lfo, lg);
      this.master.gain.linearRampToValueAtTime(volume, this.ctx.currentTime + 4);
      this.running = true;
    }
    setVolume(v) { if (this.master && this.ctx) this.master.gain.linearRampToValueAtTime(v, this.ctx.currentTime + 0.5); }
    setPhaseFreq(phase) {
      // Subtle frequency shift per phase to reinforce deepening

    }

      this.nodes.forEach(n => { try { n.stop && n.stop(); n.disconnect && n.disconnect(); } catch(e) {} });
      try { await this.ctx.close(); } catch(e) {}
      this.ctx = null; this.master = null; this.nodes = []; this.running = false;
    }
  }

  function hexToRgb(hex) {
    let h = hex.replace('#','');
    if (h.length === 3) h = h[0]+h[0]+h[1]+h[1]+h[2]+h[2];
    const r = parseInt(h.slice(0,2),16), g = parseInt(h.slice(2,4),16), b = parseInt(h.slice(4,6),16);
    return `${Number.isFinite(r)?r:128},${Number.isFinite(g)?g:128},${Number.isFinite(b)?b:128}`;
  }

  function renderTunnel(ctx, W, H, t, color) {
    const cx = W/2, cy = H/2, rgb = hexToRgb(color), maxR = Math.hypot(cx,cy)*1.1, RINGS = 18;
    ctx.fillStyle = '#000'; ctx.fillRect(0,0,W,H);
    for (let i = RINGS; i >= 0; i--) {
      const frac = ((i/RINGS) + (t*0.18)) % 1, r = frac*maxR, alpha = (1-frac)*0.65+0.05;
      ctx.beginPath(); ctx.arc(cx,cy,r,0,Math.PI*2);
      ctx.strokeStyle = `rgba(${rgb},${alpha})`; ctx.lineWidth = 1.5+(1-frac)*3; ctx.stroke();
    }
    const g = ctx.createRadialGradient(cx,cy,0,cx,cy,maxR*0.15);
    g.addColorStop(0,`rgba(${rgb},0.7)`); g.addColorStop(1,`rgba(${rgb},0)`);
    ctx.beginPath(); ctx.arc(cx,cy,maxR*0.15,0,Math.PI*2); ctx.fillStyle = g; ctx.fill();
  }

  function renderMandala(ctx, W, H, t, color) {
    const cx = W/2, cy = H/2, petals = 8, rgb = hexToRgb(color);
    ctx.fillStyle = 'rgba(0,0,0,0.06)'; ctx.fillRect(0,0,W,H);
    ctx.strokeStyle = `rgba(${rgb},0.7)`; ctx.shadowColor = color; ctx.shadowBlur = 12;
    for (let layer = 0; layer < 4; layer++) {
      const R = (layer+1)*Math.min(W,H)*0.1;
      const spin = t*(layer%2===0 ? 0.2 : -0.15) + layer*0.4;
      ctx.globalAlpha = 0.6-layer*0.1; ctx.lineWidth = 1.8-layer*0.3;
      for (let p = 0; p < petals; p++) {
        const base = (p/petals)*Math.PI*2 + spin, tip = base + Math.PI/petals;
        ctx.beginPath(); ctx.moveTo(cx,cy);
        ctx.quadraticCurveTo(cx+R*1.5*Math.cos(base+0.3),cy+R*1.5*Math.sin(base+0.3),cx+R*Math.cos(tip),cy+R*Math.sin(tip));
        ctx.stroke();
      }
      ctx.beginPath(); ctx.arc(cx,cy,R*0.4,0,Math.PI*2); ctx.stroke();
    }
    ctx.globalAlpha = 1; ctx.shadowBlur = 0;
  }

  function renderColorWash(ctx, W, H, t, color) {
    const hue = (t*12 + parseInt(color.replace('#',''),16)%360) % 360, hue2 = (hue+40)%360;
    const grad = ctx.createRadialGradient(W*0.4,H*0.4,0,W/2,H/2,Math.max(W,H)*0.7);
    grad.addColorStop(0,`hsla(${hue},70%,35%,1)`);

    grad.addColorStop(1,`hsla(${(hue+180)%360},50%,8%,1)`);
    ctx.fillStyle = grad; ctx.fillRect(0,0,W,H);
    const vig = ctx.createRadialGradient(W/2,H/2,H*0.2,W/2,H/2,H*0.75);

  function renderCanvasPulse(ctx, W, H, t, color) {
    const cx = W/2, cy = H/2, rgb = hexToRgb(color), maxR = Math.min(W,H)*0.46;
    ctx.fillStyle = '#000'; ctx.fillRect(0,0,W,H);
    for (let i = 0; i < 6; i++) {
      const phase = (t*0.45 + i/6) % 1, r = phase*maxR, alpha = (1-phase)*0.7;
      ctx.beginPath(); ctx.arc(cx,cy,r,0,Math.PI*2);
      ctx.strokeStyle = `rgba(${rgb},${alpha})`; ctx.lineWidth = 2+(1-phase)*5; ctx.stroke();
    }
    ctx.beginPath(); ctx.arc(cx,cy,10+4*Math.sin(t*3),0,Math.PI*2);
    ctx.fillStyle = `rgba(${rgb},0.85)`; ctx.fill();
  }

  function parseScript(text) {
    const phases = { settling: '', induction: '', deepener: '', work: '', emergence: '' };
    const markers = [
      { key: 'settling', rx: /===\s*SETTLING\s*===/i },
      { key: 'induction', rx: /===\s*INDUCTION\s*===/i },
      { key: 'deepener', rx: /===\s*DEEPENER\s*===/i },
      { key: 'work', rx: /===\s*WORK\s*===/i },
      { key: 'emergence', rx: /===\s*EMERGENCE\s*===/i },
    ];
    const positions = markers.map(m => { const match = text.match(m.rx); return { key: m.key, index: match ? match.index + match[0].length : -1, start: match ? match.index : -1 }; });
    const found = positions.filter(p => p.index !== -1).sort((a,b) => a.index - b.index);
    for (let i = 0; i < found.length; i++) {
      const cur = found[i];
      const end = i + 1 < found.length ? found[i+1].start : text.length;
      phases[cur.key] = text.substring(cur.index, end).trim();
    }
    return phases;
  }

  async function startSession() {

    if (voiceList.length === 0) {

    }

    state.fullScript = scriptText;
    state.parsedScript = parseScript(scriptText);

      } catch(e) {
        notify.info('Audio drone unavailable — tap the screen to enable audio, then try again.');
      }
    }

    // Session time tracker
    state._timeInterval = setInterval(() => {
      if (state.paused) return;
      const elapsed = Math.round((Date.now() - state._sessionStartTime) / 1000);
      const mins = Math.floor(elapsed / 60);
      const secs = elapsed % 60;
      const timeEl = document.getElementById('sessionTime');
      if (timeEl) timeEl.textContent = `${mins}:${String(secs).padStart(2,'0')} elapsed`;
    }, 1000);

    const phases = ['settling','induction','deepener','work','emergence'];
    const phaseLabels = { settling: 'settling', induction: 'going inward', deepener: 'deeper', work: method.category === 'demonstration' ? 'the effect' : 'transformation', emergence: 're-alerting' };

    for (let pi = 0; pi < phases.length; pi++) {
      const phase = phases[pi];
      if (state.abortFlag) break;

      if (!phaseText || phaseText.trim().length < 10) continue;

      // Update phase state
      state._currentPhase = phase;
      document.getElementById('phaseIndicator').textContent = phaseLabels[phase];

      // Update phase dots
      document.querySelectorAll('.phase-dot').forEach((d, i) => {
        if (i < pi) d.classList.add('done');
        d.classList.toggle('active', i === pi);
      });

      // Update progress bar
      const pct = Math.round((pi / phases.length) * 100);
      document.getElementById('sessionProgressFill').style.width = pct + '%';

      // Modulate drone frequency per phase
      if (state.droneEnabled && drone.running) drone.setPhaseFreq(phase);

      const cleanText = phaseText.replace(/^===\s*\w+\s*===\s*/i, '').trim();
      await speakScript(cleanText);
    }

    // Complete progress bar
    document.getElementById('sessionProgressFill').style.width = '100%';
    document.querySelectorAll('.phase-dot').forEach(d => { d.classList.remove('active'); d.classList.add('done'); });

    if (!state.abortFlag) finishSession(false);
  }

  // Interruptible sleep — checks abortFlag & paused state in 200ms chunks so
  // pause/end during a [PAUSE n] marker responds quickly instead of waiting
  // for the whole sleep to finish.
  async function interruptibleSleep(totalMs, bypassAbort = false) {
    const CHUNK = 200;
    let elapsed = 0;
    while (elapsed < totalMs) {
      if (!bypassAbort && state.abortFlag) return;
      while (state.paused) { await sleep(CHUNK); }
      const remaining = totalMs - elapsed;
      await sleep(Math.min(CHUNK, remaining));
      elapsed += CHUNK;
    }
  }

  async function speakScript(text, bypassAbort = false) {
    // Sanitize: strip any malformed [pause...] or [PAUSE...] markers that
    // wouldn't match our canonical regex. Prevents TTS from reading "bracket
    // pause bracket" aloud.
    const canonical = text.replace(/\[\s*pause[^\]]*\]/gi, (m) => {
      const num = m.match(/\d+/);
      return num ? `[PAUSE ${num[0]}]` : '';
    });

    const parts = canonical.split(/\[PAUSE\s+(\d+)\]/i);
    const display = document.getElementById('textDisplay');
    for (let i = 0; i < parts.length; i++) {
      if (!bypassAbort && state.abortFlag) return;
      while (state.paused) await sleep(200);
      if (i % 2 === 0) {
        const segment = parts[i].trim();
        if (!segment) continue;
        const chunks = segment.split(/(?<=[.!?])\s+/).filter(s => s.trim());
        for (const chunk of chunks) {
          if (!bypassAbort && state.abortFlag) return;
          while (state.paused) await sleep(200);
          display.style.opacity = 0;
          await sleep(350);
          display.textContent = chunk;
          display.style.opacity = 0.5;
          await speak(chunk);
        }
      } else {
        const seconds = parseInt(parts[i], 10) || 4;
        await interruptibleSleep(seconds * 1000, bypassAbort);
      }
    }
  }

  function speak(text) {
    return new Promise(resolve => {
      if (!('speechSynthesis' in window)) { setTimeout(resolve, text.length * 80); return; }
      // Chrome bug workaround: cancel + re-init if synthesis has been going too long
      if (speechSynthesis.speaking && !speechSynthesis.paused) {
        speechSynthesis.cancel();
        // Brief gap to let Chrome reset its internal state
        setTimeout(() => _doSpeak(text, resolve), 50);
      } else {
        _doSpeak(text, resolve);
      }
    });
  }
  function _doSpeak(text, resolve) {
    const u = new SpeechSynthesisUtterance(text);
    u.voice = state.voice || getSelectedVoice();
    u.rate = state.speechRate; u.pitch = 0.85; u.volume = 1.0;
    let settled = false;
    const done = () => { if (!settled) { settled = true; resolve(); } };
    u.onend = done;
    u.onerror = done;
    speechSynthesis.speak(u);
    // Watchdog: if TTS hasn't fired onend within a generous timeout, force-resolve
    const maxWait = Math.max(8000, text.length * 200);
    setTimeout(() => {
      if (!settled && !speechSynthesis.speaking) done();
    }, maxWait);
    // Hard ceiling to prevent infinite hangs
    setTimeout(done, Math.max(15000, text.length * 250));
  }
  const sleep = ms => new Promise(r => setTimeout(r, ms));

  async function finishSession(wasEmergency = false) {
    clearInterval(state._timeInterval);
    await drone.stop(); stopVisuals(); speechSynthesis.cancel();
    document.body.classList.remove('in-session');
    document.getElementById('session-screen').style.display = 'none';
    document.getElementById('post-screen').style.display = 'block';

    // Build post-session summary
    const elapsed = state._sessionStartTime ? Math.round((Date.now() - state._sessionStartTime) / 1000) : 0;
    const mins = Math.floor(elapsed / 60), secs = elapsed % 60;
    const persona = getPersona(), method = getMethod();
    const summaryEl = document.getElementById('postSummary');
    summaryEl.innerHTML = `
      <h3>session summary</h3>
      <div class="post-row"><span class="post-label">guide</span><span class="post-value">${escapeHtml(persona.name)}</span></div>
      <div class="post-row"><span class="post-label">method</span><span class="post-value">${escapeHtml(method.name)}</span></div>
      <div class="post-row"><span class="post-label">duration</span><span class="post-value">${mins}:${String(secs).padStart(2,'0')}</span></div>
      <div class="post-row"><span class="post-label">completion</span><span class="post-value">${wasEmergency ? 'ended early (re-alerted)' : 'completed'}</span></div>
    `;

    // Show trigger/suggestion reminder if relevant
    const triggerBox = document.getElementById('postTriggerBox');
    if (state.craftedSuggestion && state.craftedSuggestion.trigger && state.craftedSuggestion.trigger !== 'ongoing state') {
      triggerBox.innerHTML = `
        <div class="post-trigger">
          <div class="trigger-label">your trigger</div>

          <div style="font-size:0.75rem; color:var(--text-dim); margin-top:6px;">→ ${escapeHtml(state.craftedSuggestion.coreResponse || '')}</div>
        </div>`;
    } else if (state.craftedSuggestion && state.craftedSuggestion.coreResponse) {
      triggerBox.innerHTML = `
        <div class="post-trigger">
          <div class="trigger-label">suggestion installed</div>
          <div class="trigger-text">${escapeHtml(state.craftedSuggestion.coreResponse)}</div>
        </div>`;
    } else {
      triggerBox.innerHTML = '';
    }

    // For demonstration methods, remind user of the test
    if (method.category === 'demonstration' && !wasEmergency) {
      triggerBox.innerHTML += `<div class="post-trigger" style="margin-top:8px; border-color: var(--accent-purple);">
        <div class="trigger-label" style="color:var(--accent-purple);">test now</div>
        <div class="trigger-text" style="font-size:0.85rem;">Try the ${escapeHtml(method.name)} effect now. The suggestion should still be active.</div>
      </div>`;
    }

    // Emergency cooldown (30s after emergency end)
    if (wasEmergency) {
      state._cooldownUntil = Date.now() + 30000;
    }
  }
